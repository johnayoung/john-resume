Keyed by normalized URL so repeat renders (other output formats) reuse the same numbers. */ -}} {{- $cite := 0 -}} {{- if and $isExternal (eq .Page.Section "blog") (eq .Page.Kind "page") (not (strings.Contains .Destination "jyoung.dev")) -}} {{- $key := strings.TrimSuffix "/" .Destination -}} {{- $nums := .Page.Store.Get "citeNums" | default dict -}} {{- with index $nums $key -}} {{- $cite = . -}} {{- else -}} {{- $cite = add (len $nums) 1 -}} {{- .Page.Store.SetInMap "citeNums" $key $cite -}} {{- .Page.Store.SetInMap "citeTitles" (string $cite) (.Text | plainify | chomp) -}} {{- .Page.Store.SetInMap "citeHrefs" (string $cite) $key -}} {{- end -}} {{- end -}} {{ .Text | safeHTML }} {{- if $cite }}{{ $cite }}{{ end -}} {{- /* no trailing newline: it would render as a space after every link */ -}}
